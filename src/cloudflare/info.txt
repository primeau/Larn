https://developers.cloudflare.com/workers/get-started/guide/

to install:
npm create cloudflare@latest
sudo npm install -g wrangler@latest

*** deploy must be from larn-local

View your deployed application at https://broadcast.larn.workers.dev

Navigate to the new directory: cd broadcast
Run the development server: 
  % npm run start
Deploy your application:
  % npm run deploy
Tail logs: 
  % npx wrangler tail
tail remote logs:
  % npx wrangler --config ../../larn-local/cloudflare/broadcast/wrangler.toml tail
Test dev:
  % npx wrangler dev --test-scheduled 
Test scheduled event: 
  # http://localhost:8787/__scheduled?cron=*+*+*+*+*

list R2 buckets:
 % npx wrangler r2 bucket list

create D1 database
  % npx wrangler d1 create larn
create tables
  % npx wrangler d1 execute larn --local --file=./schema.sql
execute sql
  % wrangler d1 execute larn --local --command "SELECT count (*) from TABLE"  
export local tables
  % wrangler d1 export larn --local --table=<TABLE_NAME> --output=./dump.sql --config ../../larn/src/cloudflare/broadcast/wrangler.toml
import to remote table 
  % wrangler d1 execute larn --remote --file=./dump.sql --config <CONFIG_FILE_PATH>
list all tables
  % wrangler d1 execute larn --local --command "PRAGMA table_list"
